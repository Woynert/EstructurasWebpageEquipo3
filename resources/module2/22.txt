class Pila:

    def __init__(self):
        self.items=[]

    def push(self, x):
        self.items.append(x)

    def pop(self):
        try:
            return self.items.pop()
        except IndexError:
            raise ValueError("La pila está vacía")

    def peek(self):
        try:
            return self.items[-1]
        except IndexError:
            raise ValueError("La pila está vacía")

    def isEmpty(self):
        return self.items == []

    def clear(self):
        self.__init__()

    def size(self):
        try:
            return len(self.items)
        except IndexError:
            raise ValueError("La pila está vacía")
